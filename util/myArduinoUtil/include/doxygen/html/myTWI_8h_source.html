40 /*
41  * According to the AMG88** data sheet the SCL clock frequency (f_SCL) has
42  * min = 0 kHz and max = 400 kHZ
43  *
44  * According to the ATmega328p data sheet the SCL clock frequency is
45  * f_SCL = CPU_ClockFrequency / (16 + 2 * TWBR * PrescalerValue)
46  * where TWBR is the value of the TWI Bit Rate Register
47  * and the PrescaleValue (in {1, 4, 16, 64}) can be set through the TWSR (TWI Status Register)
48  *
49  * Assuming we want to work with a f_SCL of 100kHz
50  * => TWBR * PrescalerValue = 72
51  * => TWBR = 18 = 0b00010010
52  * => PrescalerValue: TWPS1 = 0, TWPS0 = 1 (both bits are part of byte TWSR)
53  * #define TWI_PRESCALER_VALUE 0x01 // PrescalerValue: 4
54  * #define TWI_BIT_RATE 0x12 // Decimal: 18
55  *
56  *
57  * In General f_SCL = x (known)
58  * TWBR * PrescalerValue =: y (unknown)
59  * x = CPU_ClockFrequency / (16 + 2y)
60  * <=> 16x + 2xy = CPU_ClockFrequency
61  * <=> 2xy = CPU_ClockFrequency - 16x
62  * <=> y = (CPU_ClockFrequency - 16x) / 2x
63  *
64  * x = 400kHz
65  * => y = 12 (= TWBR * PrescalerValue)
66 */
67 
68 #define TWI_PRESCALER_VALUE 0x00 // TWPS1 = 0 and TWPS0 = 0 means PrescalerValue of 1
69 #define TWI_BIT_RATE 0x0C // Decimal: 12
